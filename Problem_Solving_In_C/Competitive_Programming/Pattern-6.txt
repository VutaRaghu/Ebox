#include<stdio.h>
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        if(i==1)
        {
            for(j=1;j<=n;j++)
            {
                if((j==n/2)||(j==(n/2)+1))
                {
                    printf("C");
                }
                else
                {
                    printf(".");
                }
            }
        }
        
        else if(i<=n/2)
        {
            for(j=1;j<=n;j++)
            {
                if((j==(n/2)-(i-1))||(j==(n/2)+i))
                {
                    printf("S");
                }
                else
                {
                    printf(".");
                }
            }
        }
        else
        {
            for(j=1;j<=n;j++)
            {
                if((j==(n/2)-(n-i))||(j==(n/2)+(n-i+1)))
                {
                    printf("K");
                }
                else
                {
                    printf(".");
                }
            }
        }
        printf("\n");
    }
}
