#include<stdio.h>
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            if(i==1)
            {
                printf("C");
            }
            else if(i==n)
            {
                printf("K");
            }
            else if((j==1||j==n)&&(i!=1)&(i!=n))
            {
                printf("S");
            }
            else
            {
                printf("*");
            }
        }
        printf("\n");
    }
}
